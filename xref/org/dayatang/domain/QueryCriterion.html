4   /**
5    * 查询条件接口
6    *
7    * @author yyang
8    */
9   public interface QueryCriterion {
10  
11      static final String ROOT_ALIAS = "rootEntity";
12  
13      /**
14       * 执行AND操作，返回代表两个QueryCriterion的“与”操作结果的一个新的QueryCriterion
15       *
16       * @param criterion 另一个QueryCriterion
17       * @return 当前对象与criterion的“与”操作的结果
18       */
19      QueryCriterion and(QueryCriterion criterion);
20  
21      /**
22       * 执行OR操作，返回代表两个QueryCriterion的“或”操作结果的一个新的QueryCriterion
23       *
24       * @param criterion 另一个QueryCriterion
25       * @return 当前对象与criterion的“或”操作的结果
26       */
27      QueryCriterion or(QueryCriterion criterion);
28  
29      /**
30       * 执行NOT操作，返回代表当前对象的“非”操作的一个新的QueryCriterion
31       * @return 当前对象的“非”操作的结果
32       */
33      QueryCriterion not();
34  
35      /**
36       * 是否空条件，即EmptyCriterion的实例
37       * @return 如果是空条件就返回true，否则返回false
38       */
39      boolean isEmpty();
40  
41      /**
42       * 转换成JPQL字符串
43       * @return 查询字符串
44       */
45      String toQueryString();
46  
47      /**
48       * 获得查询参数
49       * @return 查询的参数集
50       */
51      NamedParameters getParameters();
52  
53  
54  }
