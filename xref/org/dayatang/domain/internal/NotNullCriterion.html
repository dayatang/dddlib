1   package org.dayatang.domain.internal;
2   
3   import org.apache.commons.lang3.builder.EqualsBuilder;
4   import org.apache.commons.lang3.builder.HashCodeBuilder;
5   import org.dayatang.domain.NamedParameters;
6   
7   /**
8    * 判断某个属性值是否为不null的查询条件
9    * @author yyang
10   */
11  public class NotNullCriterion extends BasicCriterion {
12  
13      public NotNullCriterion(String propName) {
14          super(propName);
15      }
16  
17      @Override
18      public String toQueryString() {
19          return getPropNameWithAlias() + " is not null";
20      }
21  
22      @Override
23      public NamedParameters getParameters() {
24          return NamedParameters.create();
25      }
26  
27      @Override
28      public boolean equals(final Object other) {
29          if (this == other) {
30              return true;
31          }
32          if (!(other instanceof NotNullCriterion)) {
33              return false;
34          }
35          NotNullCriterion that = (NotNullCriterion) other;
36          return new EqualsBuilder()
37                  .append(this.getPropName(), that.getPropName())
38                  .isEquals();
39      }
40  
41      @Override
42      public int hashCode() {
43          return new HashCodeBuilder(17, 37).append(getPropName()).toHashCode();
44      }
45  
46      @Override
47      public String toString() {
48          return getPropName() + " is not null";
49      }
50  
51  }
