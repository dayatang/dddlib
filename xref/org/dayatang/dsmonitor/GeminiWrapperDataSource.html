1   package org.dayatang.dsmonitor;
2   
3   import org.dayatang.utils.Assert;
4   
5   import javax.sql.DataSource;
6   import java.sql.Connection;
7   import java.sql.SQLException;
8   import java.util.Collections;
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  public class GeminiWrapperDataSource extends DelegatingDataSource {
13  
14      private Set<ConnectionMonitor> monitors = new HashSet<ConnectionMonitor>();
15  
16      public GeminiWrapperDataSource(DataSource targetDataSource, Set<ConnectionMonitor> monitors) {
17          super(targetDataSource);
18          setMonitors(monitors);
19      }
20  
21      public Connection getConnection() throws SQLException {
22          return buildConnection(super.getConnection());
23      }
24  
25      public Connection getConnection(String username, String password)
26              throws SQLException {
27          return buildConnection(super.getConnection(username, password));
28      }
29  
30      public Set<ConnectionMonitor> getMonitors() {
31          return Collections.unmodifiableSet(monitors);
32      }
33  
34      public final void setMonitors(Set<ConnectionMonitor> monitors) {
35          if (monitors != null) {
36              this.monitors = monitors;
37          }
38      }
39  
40      public void addMonitor(ConnectionMonitor monitor) {
41          Assert.notNull(monitor, "Monitor to add cannot be null!");
42          monitors.add(monitor);
43      }
44  
45      public void removeMonitor(ConnectionMonitor monitor) {
46          Assert.notNull(monitor, "Monitor to add cannot be null!");
47          monitors.remove(monitor);
48      }
49  
50      private Connection buildConnection(Connection connection) throws SQLException {
51          if (connection == null) {
52              return null;
53          }
54          return new GeminiConnection(connection, monitors);
55      }
56  
57  }
