11  public abstract class DeveloperTestSuite {
12  
13  	private static final Logger logger = LoggerFactory
14  			.getLogger(DeveloperTestSuite.class);
15  
16  	private static String initSQLFile = PropertiesUtil.INIT_SQL_FILE;
17  
18  	private static DBManager dbMgr;
19  
20  	public static DBManager getDbMgr() {
21  		if (dbMgr == null) {
22  			dbMgr = InstanceFactory.getInstance(DBManager.class);
23  		}
24  		return dbMgr;
25  	}
26  
27  	public static void setDbMgr(DBManager dbMgr) {
28  		DeveloperTestSuite.dbMgr = dbMgr;
29  	}
30  
31  	@BeforeClass
32  	public static void startSuite() throws Exception {
33  		// 我会在所有测试类运行之前执行
34  		logger.info("============这里是测试套件=============");
35  		// 该语句会执行SQL 重新插入数据
36  		Thread.sleep(3000);
37  
38  		dbMgr.cleanDatabase();
39  
40  		dbMgr.executeSQL(initSQLFile);
41  
42  	}
43  
44  	@AfterClass
45  	public static void endSuite() {
46  		// 我会在所有测试类运行之后执行
47  		logger.info("============测试套件执行完毕=============");
48  	}
49  }
