1   package org.dayatang.observer;

3   import org.dayatang.domain.AbstractEntity;
4   
5   import javax.persistence.MappedSuperclass;
6   import java.util.List;
7   
8   @SuppressWarnings("unchecked")
9   @MappedSuperclass
10  public abstract class AbstractSubjectEntity extends AbstractEntity implements
11          Subject {
12  
13      /**
14       *
15       */
16      private static final long serialVersionUID = 8696877724719553088L;
17  
18      /**
19       * 获得主题标识。默认返回主题的类名。
20       *
21       * @return
22       */
23      public String getSubjectKey() {
24          return getClass().getName();
25      }
26  
27      @Override
28      public List<Observer> getObservers() {
29          // 每次查找最新的观察者，将来考虑加入缓存机制
30          return Observer.findBySubject(this);
31  
32      }
33  
34      @Override
35      public void notifyObservers() {
36          List<Observer> observers = getObservers();
37          for (Observer observer : observers) {
38              observer.process(this);
39          }
40      }
41  
42  }
