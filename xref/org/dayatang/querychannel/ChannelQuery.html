16  package org.dayatang.querychannel;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  import org.dayatang.domain.BaseQuery;
22  import org.dayatang.domain.EntityRepository;
23  import org.dayatang.domain.QueryParameters;
24  import org.dayatang.utils.Assert;
25  import org.dayatang.utils.Page;
26  
27  /**
28   * 查询通道查询
29   *
30   * @author yyang
31   * @param <E> 查询类型
32   */
33  public abstract class ChannelQuery<E extends ChannelQuery> {
34  
35      protected EntityRepository repository;
36      private BaseQuery query;
37      private int pageIndex;
38      //private int pageSize = Page.DEFAULT_PAGE_SIZE;
39  
40  
41      public ChannelQuery(EntityRepository repository) {
42          this.repository = repository;
43      }
44  
45      public void setQuery(BaseQuery query) {
46          this.query = query;
47      }
48  
49      /**
50       * 获取查询参数
51       *
52       * @return 查询参数
53       */
54      public QueryParameters getParameters() {
55          return query.getParameters();
56      }
57  
58      /**
59       * 设置定位命名参数（数组方式）
60       *
61       * @param parameters 要设置的参数
62       * @return 该对象本身
63       */
64      public E setParameters(Object... parameters) {
65          query.setParameters(parameters);
66          return (E) this;
67      }
68  
69      /**
70       * 设置定位参数（列表方式）
71       *
72       * @param parameters 要设置的参数
73       * @return 该对象本身
74       */
75      public E setParameters(List<Object> parameters) {
76          query.setParameters(parameters);
77          return (E) this;
78      }
79  
80      /**
81       * 设置命名参数（Map形式，Key是参数名称，Value是参数值）
82       *
83       * @param parameters 要设置的参数
84       * @return 该对象本身
85       */
86      public E setParameters(Map<String, Object> parameters) {
87          query.setParameters(parameters);
88          return (E) this;
89      }
90  
91      /**
92       * 添加一个命名参数，Key是参数名称，Value是参数值。
93       *
94       * @param key 命名参数名称
95       * @param value 参数值
96       * @return 该对象本身
97       */
98      public E addParameter(String key, Object value) {
99          query.addParameter(key, value);
100         return (E) this;
101     }
102 
103     /**
104      * 针对分页查询，获取firstResult。 firstResult代表从满足查询条件的记录的第firstResult + 1条开始获取数据子集。
105      *
106      * @return firstResult的设置值，
107      */
108     public int getFirstResult() {
109         return query.getFirstResult();
110     }
111 
112     /**
113      * 针对分页查询，设置firstResult。 firstResult代表从满足查询条件的记录的第firstResult + 1条开始获取数据子集。
114      *
115      * @param firstResult 要设置的firstResult值。
116      * @return 该对象本身
117      */
118     public E setFirstResult(int firstResult) {
119         Assert.isTrue(firstResult >= 0, "First result must be greater than 0!");
120         query.setFirstResult(firstResult);
121         return (E) this;
122     }
123 
124     /**
125      * 获取当前页码（0为第一页）
126      *
127      * @return 当前页码
128      */
129     public int getPageIndex() {
130         return pageIndex;
131     }
132 
133     /**
134      * 获取每页记录数
135      *
136      * @return 每页记录数
137      */
138     public int getPageSize() {
139         return query.getMaxResults();
140     }
141 
142     /**
143      * 设置每页记录数
144      *
145      * @param pageSize 每页记录数
146      * @return 该对象本身
147      */
148     public E setPageSize(int pageSize) {
149         Assert.isTrue(pageSize > 0, "Page size must be greater than 0!");
150         query.setMaxResults(pageSize);
151         return (E) this;
152     }
153 
154     /**
155      * 设置分页信息
156      *
157      * @param pageIndex 要设置的页码
158      * @param pageSize 要设置的页大小
159      * @return 该对象本身
160      */
161     public E setPage(int pageIndex, int pageSize) {
162         Assert.isTrue(pageIndex >= 0, "Page index must be greater than or equals to 0!");
163         Assert.isTrue(pageSize > 0, "Page index must be greater than 0!");
164         this.pageIndex = pageIndex;
165         query.setMaxResults(pageSize);
166         query.setFirstResult(Page.getStartOfPage(pageIndex, pageSize));
167         return (E) this;
168     }
169 
170     /**
171      * 返回查询结果数据页。
172      *
173      * @param <T> 查询结果的列表元素类型
174      * @return 查询结果。
175      */
176     public <T> List<T> list() {
177         return query.list();
178     }
179 
180     /**
181      * 返回查询结果数据页。
182      *
183      * @param <T> 查询结果的列表元素类型
184      * @return 查询结果。
185      */
186     public <T> Page<T> pagedList() {
187         return new Page<T>(query.getFirstResult(), queryResultCount(), 
188                 query.getMaxResults(), query.list());
189     }
190      
191     /**
192      * 返回单条查询结果。
193      *
194      * @param <T> 查询结果的类型
195      * @return 查询结果。
196      */
197     public <T> T singleResult() {
198         return (T) query.singleResult();
199     }
200 
201     /**
202      * 获取符合查询条件的记录总数
203      *
204      * @return 符合查询条件的记录总数
205      */
206     public long queryResultCount() {
207         CountQueryStringBuilder builder = new CountQueryStringBuilder(getQueryString());
208         if (builder.containsGroupByClause()) {
209             List rows = createBaseQuery(builder.removeOrderByClause())
210                     .setParameters(query.getParameters()).list();
211             return rows == null ? 0 : rows.size();
212         } else {
213             Number result = (Number) createBaseQuery(builder.buildQueryStringOfCount())
214                     .setParameters(query.getParameters()).singleResult();
215             return result.longValue();
216         }
217     }
218 
219     /**
220      * 获得当前查询对应的查询字符串
221      * @return 当前查询对应的查询字符串
222      */
223     protected abstract String getQueryString();
224 
225     protected abstract BaseQuery createBaseQuery(String queryString);
226 }
