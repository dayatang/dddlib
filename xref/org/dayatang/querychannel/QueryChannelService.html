1   package org.dayatang.querychannel;
2   
3   
4   import org.dayatang.utils.Page;
5   
6   import java.io.Serializable;
7   import java.util.List;
8   
9   /**
10   * 查询通道接口。可以针对仓储进行分页查询。
11   */
12  public interface QueryChannelService extends Serializable {
13  
14      /**
15       * 创建JPQL查询
16       *
17       * @param jpql JPQL语句
18       * @return 一个JPQL查询
19       */
20      ChannelQuery createJpqlQuery(String jpql);
21  
22      /**
23       * 创建命名查询
24       *
25       * @param queryName 命名查询的名字
26       * @return 一个命名查询
27       */
28      ChannelQuery createNamedQuery(String queryName);
29  
30      /**
31       * 创建原生SQL查询
32       *
33       * @param sql SQL语句
34       * @return 一个原生SQL查询
35       */
36      ChannelQuery createSqlQuery(String sql);
37  
38      /**
39       * 执行查询，返回符合条件的结果列表
40       *
41       * @param query 要执行的查询
42       * @param <T> 返回结果元素类型
43       * @return 符合查询条件的结果列表
44       */
45      <T> List<T> list(ChannelQuery query);
46  
47      /**
48       * 执行查询，分页返回符合条件的结果列表
49       *
50       * @param query 要执行的查询
51       * @param <T> 返回结果元素类型
52       * @return 符合查询条件的结果页
53       */
54      <T> Page<T> pagedList(ChannelQuery query);
55  
56      /**
57       * 执行查询，返回符合条件的单个实体
58       *
59       * @param query 要执行的查询
60       * @param <T> 返回结果类型
61       * @return 符合查询条件的单个结果
62       */
63      <T> T getSingleResult(ChannelQuery query);
64  
65      /**
66       * 获取查询结果的总数。
67       *
68       * @param query 要执行的查询
69       * @return 符合查询条件的结果的总数
70       */
71      long getResultCount(ChannelQuery query);
72  
73  }
