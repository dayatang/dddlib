1   package org.dayatang.datasource4saas.dsregistry;
2   
3   import org.dayatang.utils.Assert;
4   import org.dayatang.utils.Slf4JLogger;
5   
6   import javax.sql.DataSource;
7   import java.util.Date;
8   
9   /**
10   * 当还没有对应于指定的租户的数据源时，即时创建一个。
11   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
12   *
13   */
14  public class JdbcDataSourceRegistry extends MapBasedDataSourceRegistry {
15  
16  	private static final Slf4JLogger LOGGER = Slf4JLogger.getLogger(JdbcDataSourceRegistry.class);
17  	
18  	private DataSourceCreator dataSourceCreator;
19  
20  	public DataSourceCreator getDataSourceCreator() {
21  		Assert.notNull(dataSourceCreator, "Data Source Creator is null!");
22  		return dataSourceCreator;
23  	}
24  
25  	public void setDataSourceCreator(DataSourceCreator dataSourceCreator) {
26  		this.dataSourceCreator = dataSourceCreator;
27  	}
28  
29  	public JdbcDataSourceRegistry(DataSourceCreator dataSourceCreator) {
30  		this.dataSourceCreator = dataSourceCreator;
31  	}
32  
33  	@Override
34  	protected DataSource findOrCreateDataSourceForTenant(String tenant) {
35  		Assert.notBlank(tenant, "Tenant is null or blank!");
36  		DataSource result = getDataSourceCreator().createDataSourceForTenant(tenant);
37  		Date now = new Date();
38  		recordLastAccessTimeOfTenant(tenant, now);
39  		LOGGER.debug("Create data source for tenant '{}' at {}", tenant, now);
40  		return result;
41  	}
42  }
