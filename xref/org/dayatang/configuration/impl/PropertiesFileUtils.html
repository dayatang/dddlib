1   package org.dayatang.configuration.impl;

3   import org.dayatang.configuration.ConfigurationException;
4   
5   import java.io.UnsupportedEncodingException;
6   import java.util.Hashtable;
7   import java.util.Map.Entry;
8   import java.util.Properties;
9   
10  /**
11   * 解决Java Properties文件的编码问题，避免iso8859-1到utf8的相互转义
12   * 
13   * @author jzhai
14   */
15  public class PropertiesFileUtils {
16  	public static final String ISO_8859_1 = "iso8859-1";
17  	private String encoding = "UTF-8";
18  
19  	public PropertiesFileUtils(String encoding) {
20  		this.encoding = encoding;
21  	}
22  
23  	/**
24  	 * 将Properties文件，其中含有原生的UTF8等编码字符，转换为编码正确的Hashtable
25  	 */
26  	public Hashtable<String, String> rectifyProperties(Properties p) {
27  		if (p == null) {
28  			return null;
29  		}
30  		Hashtable<String, String> ret = new Hashtable<String, String>();
31  		for (Entry<Object, Object> e : p.entrySet()) {
32  			String key = (String) e.getKey();
33  			String value = (String) e.getValue();
34  			ret.put(key, StringPropertyReplacer.replaceProperties(rectifyStr(value)));
35  		}
36  		return ret;
37  	}
38  
39  	/**
40  	 * 将编码正确的Hashtable, 转换为待写入Properties文件的对象
41  	 */
42  	public Properties unRectifyProperties(Hashtable<String, String> h) {
43  		if (h == null) {
44  			return null;
45  		}
46  		Properties ret = new Properties();
47  		for (Entry<String, String> e : h.entrySet()) {
48  			String key = e.getKey();
49  			String value = e.getValue();
50  			ret.put(key, unRectifyStr(value));
51  		}
52  		return ret;
53  	}
54  
55  	/**
56  	 * 修正Properities中原生字符串为正确的编码
57  	 */
58  	public String rectifyStr(String raw) {
59  		if (raw == null) {
60  			return raw;
61  		}
62  		String ret = raw;
63  		try {
64  			byte[] bytes = raw.getBytes(ISO_8859_1);
65  			ret = new String(bytes, encoding);
66  		} catch (UnsupportedEncodingException e) {
67  			throw new ConfigurationException("Unsupport Encoding:" + encoding, e);
68  		}
69  		return ret;
70  	}
71  
72  	/**
73  	 * 还原正确的字符串编码为ISO编码
74  	 */
75  	public String unRectifyStr(String validStr) {
76  		if (validStr == null) {
77  			return validStr;
78  		}
79  		String ret = validStr;
80  		try {
81  			byte[] bytes = validStr.getBytes(encoding);
82  			ret = new String(bytes, ISO_8859_1);
83  		} catch (UnsupportedEncodingException e) {
84  			throw new ConfigurationException("Unsupport Encoding:" + encoding, e);
85  		}
86  		return ret;
87  	}
88  }
