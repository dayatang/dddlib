18  	/** New line string constant */
19  	public static final String NEWLINE = System.getProperty("line.separator",

22  	/** File separator value */
23  	private static final String FILE_SEPARATOR = File.separator;
24  
25  	/** Path separator value */
26  	private static final String PATH_SEPARATOR = File.pathSeparator;
27  
28  	/** File separator alias */
29  	private static final String FILE_SEPARATOR_ALIAS = "/";
30  
31  	/** Path separator alias */
32  	private static final String PATH_SEPARATOR_ALIAS = ":";
33  
34  	// States used in property parsing
35  	private static final int NORMAL = 0;
36  	private static final int SEEN_DOLLAR = 1;
37  	private static final int IN_BRACKET = 2;
38  
39  	private StringPropertyReplacer() {
40  		super();
41  	}

43  	/**
44  	 * Go through the input string and replace any occurance of ${p} with the
45  	 * System.getProperty(p) value. If there is no such property p defined, then
46  	 * the ${p} reference will remain unchanged.
47  	 * 
48  	 * If the property reference is of the form ${p:v} and there is no such
49  	 * property p, then the default value v will be returned.
50  	 * 
51  	 * If the property reference is of the form ${p1,p2} or ${p1,p2:v} then the
52  	 * primary and the secondary properties will be tried in turn, before
53  	 * returning either the unchanged input, or the default value.
54  	 * 
55  	 * The property ${/} is replaced with System.getProperty("file.separator")
56  	 * value and the property ${:} is replaced with
57  	 * System.getProperty("path.separator").
58  	 * 
59  	 * @param value
60  	 *            - the string with possible ${} references
61  	 * @return the input string with all property references replaced if any. If
62  	 *         there are no valid references the input string will be returned.
63  	 */
64  	public static String replaceProperties(final String value) {
65  		return replaceProperties(value, null);
66  	}
67  
68  	public static Properties replacePropertyValues(final Properties props) {
69  		if (props == null) {
70  			return null;
71  		}
72  		Properties ret = new Properties();
73  		for (Map.Entry<Object, Object> e : props.entrySet()) {
74  			String key = (String) e.getKey();
75  			String value = (String) e.getValue();
76  			ret.put(key, replaceProperties(value));
77  		}
78  		return ret;
79  	}

81  	/**
82  	 * Go through the input string and replace any occurance of ${p} with the
83  	 * props.getProperty(p) value. If there is no such property p defined, then
84  	 * the ${p} reference will remain unchanged.
85  	 * 
86  	 * If the property reference is of the form ${p:v} and there is no such
87  	 * property p, then the default value v will be returned.
88  	 * 
89  	 * If the property reference is of the form ${p1,p2} or ${p1,p2:v} then the
90  	 * primary and the secondary properties will be tried in turn, before
91  	 * returning either the unchanged input, or the default value.
92  	 * 
93  	 * The property ${/} is replaced with System.getProperty("file.separator")
94  	 * value and the property ${:} is replaced with
95  	 * System.getProperty("path.separator").
96  	 * 
97  	 * @param string
98  	 *            - the string with possible ${} references
99  	 * @param props
100 	 *            - the source for ${x} property ref values, null means use
101 	 *            System.getProperty()
102 	 * @return the input string with all property references replaced if any. If
103 	 *         there are no valid references the input string will be returned.
104 	 * @throws java.lang.AccessControlException
105 	 *             when not authorised to retrieved system properties
106 	 */
107 	public static String replaceProperties(final String string,
108 			final Properties props) {
109 		final char[] chars = string.toCharArray();
110 		StringBuffer buffer = new StringBuffer();
111 		boolean properties = false;
112 		int state = NORMAL;
113 		int start = 0;
114 		for (int i = 0; i < chars.length; ++i) {
115 			char c = chars[i];
116 
117 			// Dollar sign outside brackets
118 			if (c == '$' && state != IN_BRACKET) {
119 				state = SEEN_DOLLAR;
120 			}
121 
122 			// Open bracket immediatley after dollar
123 			else if (c == '{' && state == SEEN_DOLLAR) {
124 				buffer.append(string.substring(start, i - 1));
125 				state = IN_BRACKET;
126 				start = i - 1;
127 			}
128 
129 			// No open bracket after dollar
130 			else if (state == SEEN_DOLLAR) {
131 				state = NORMAL;
132 			}
133 
134 			// Closed bracket after open bracket
135 			else if (c == '}' && state == IN_BRACKET) {
136 				// No content
137 				if (start + 2 == i) {
138 					buffer.append("${}"); // REVIEW: Correct?
139 				} else // Collect the system property
140 				{
141 					String value = null;
142 
143 					String key = string.substring(start + 2, i);
144 
145 					// check for alias
146 					if (FILE_SEPARATOR_ALIAS.equals(key)) {
147 						value = FILE_SEPARATOR;
148 					} else if (PATH_SEPARATOR_ALIAS.equals(key)) {
149 						value = PATH_SEPARATOR;
150 					} else {
151 						// check from the properties
152 						if (props != null) {
153 							value = props.getProperty(key);
154 						}
155 						else {
156 							value = System.getProperty(key);
157 						}
158 
159 						if (value == null) {
160 							// Check for a default value ${key:default}
161 							int colon = key.indexOf(':');
162 							if (colon > 0) {
163 								String realKey = key.substring(0, colon);
164 								if (props != null) {
165 									value = props.getProperty(realKey);
166 								}
167 								else {
168 									value = System.getProperty(realKey);
169 								}
170 
171 								if (value == null) {
172 									// Check for a composite key, "key1,key2"
173 									value = resolveCompositeKey(realKey, props);
174 
175 									// Not a composite key either, use the
176 									// specified default
177 									if (value == null) {
178 										value = key.substring(colon + 1);
179 									}
180 								}
181 							} else {
182 								// No default, check for a composite key,
183 								// "key1,key2"
184 								value = resolveCompositeKey(key, props);
185 							}
186 						}
187 					}
188 
189 					if (value != null) {
190 						properties = true;
191 						buffer.append(value);
192 					}
193 				}
194 				start = i + 1;
195 				state = NORMAL;
196 			}
197 		}
198 
199 		// No properties
200 		if (properties == false) {
201 			return string;
202 		}
203 
204 		// Collect the trailing characters
205 		if (start != chars.length) {
206 			buffer.append(string.substring(start, chars.length));
207 		}
208 		// Done
209 		return buffer.toString();
210 	}

212 	/**
213 	 * Try to resolve a "key" from the provided properties by checking if it is
214 	 * actually a "key1,key2", in which case try first "key1", then "key2". If
215 	 * all fails, return null.
216 	 * 
217 	 * It also accepts "key1," and ",key2".
218 	 * 
219 	 * @param key
220 	 *            the key to resolve
221 	 * @param props
222 	 *            the properties to use
223 	 * @return the resolved key or null
224 	 */
225 	private static String resolveCompositeKey(String key, Properties props) {
226 		String value = null;
227 
228 		// Look for the comma
229 		int comma = key.indexOf(',');
230 		if (comma > -1) {
231 			// If we have a first part, try resolve it
232 			if (comma > 0) {
233 				// Check the first part
234 				String key1 = key.substring(0, comma);
235 				if (props != null) {
236 					value = props.getProperty(key1);
237 				}
238 				else {
239 					value = System.getProperty(key1);
240 				}
241 			}
242 			// Check the second part, if there is one and first lookup failed
243 			if (value == null && comma < key.length() - 1) {
244 				String key2 = key.substring(comma + 1);
245 				if (props != null) {
246 					value = props.getProperty(key2);
247 				}
248 				else {
249 					value = System.getProperty(key2);
250 				}
251 			}
252 		}
253 		// Return whatever we've found or null
254 		return value;
255 	}
