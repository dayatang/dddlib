18  /**
19   * 通用仓储接口的Hibernate实现。
20   * <p> EntityRepositoryHibernate通过SessionProvider获取Session，以保证在当前线程和事务中
21   * 对数据库的多次访问都是由同一个Session来进行，防止出现“会话已关闭”异常。
22   *
23   * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
24   *
25   */
26  @SuppressWarnings({"unchecked"})
27  public class EntityRepositoryHibernate implements EntityRepository {

46      /*
47       * (non-Javadoc)
48       * @see org.dayatang.domain.EntityRepository#save(org.dayatang.domain.Entity)
49       */
50      @Override
51      public <T extends Entity> T save(T entity) {
52          if (entity.notExisted()) {
53              getSession().save(entity);
54              LOGGER.info("create a entity: " + entity.getClass() + "/" + entity.getId() + ".");
55              return entity;
56          }
57          getSession().update(entity);
58          LOGGER.info("update a entity: " + entity.getClass() + "/" + entity.getId() + ".");
59          return entity;
60      }

262     private Query getQuery(SqlQuery sqlQuery) {
263         SQLQuery query = getSession().createSQLQuery(sqlQuery.getSql());
264         processQuery(query, sqlQuery);
265         Class resultEntityClass = sqlQuery.getResultEntityClass();
266         if (resultEntityClass != null) {
267             query.addEntity(resultEntityClass);
268         }
269         return query;
270     }

272     /*
273      * (non-Javadoc)
274      * @see org.dayatang.domain.EntityRepository#findByExample(org.dayatang.domain.Entity, org.dayatang.domain.ExampleSettings)
275      */
276     @Override
277     public <T extends Entity, E extends T> List<T> findByExample(final E example, final ExampleSettings<T> settings) {
278         Example theExample = Example.create(example);
279         if (settings.isLikeEnabled()) {
280             theExample.enableLike(MatchMode.ANYWHERE);
281         }
282         if (settings.isIgnoreCaseEnabled()) {
283             theExample.ignoreCase();
284         }
285         if (settings.isExcludeNone()) {
286             theExample.excludeNone();
287         }
288         if (settings.isExcludeZeroes()) {
289             theExample.excludeZeroes();
290         }
291         for (String propName : settings.getExcludedProperties()) {
292             theExample.excludeProperty(propName);
293         }
294         return getSession().createCriteria(settings.getEntityClass()).add(theExample).list();
295     }

310     @Override
311     public <T extends Entity> List<T> findByProperties(Class<T> clazz, NamedParameters properties) {
312         CriteriaQuery criteriaQuery = new CriteriaQuery(this, clazz);
313         for (Map.Entry<String, Object> each : properties.getParams().entrySet()) {
314             criteriaQuery = criteriaQuery.eq(each.getKey(), each.getValue());
315         }
316         return find(criteriaQuery);
317     }

356     private void processQuery(Query query, BaseQuery originQuery) {
357         processQuery(query, originQuery.getParameters(), originQuery.getFirstResult(), 
358                 originQuery.getMaxResults());
359         fillParameters(query, originQuery.getParameters());
360         query.setFirstResult(originQuery.getFirstResult());
361         if (originQuery.getMaxResults() > 0) {
362             query.setMaxResults(originQuery.getMaxResults());
363         }
364     }
365 
366     private void processQuery(Query query, QueryParameters parameters, 
367             int firstResult, int maxResults) {
368         fillParameters(query, parameters);
369         query.setFirstResult(firstResult);
370         if (maxResults > 0) {
371             query.setMaxResults(maxResults);
372         }
373     }
374 
375     private void fillParameters(Query query, QueryParameters params) {
376         if (params == null) {
377             return;
378         }
379         if (params instanceof PositionalParameters) {
380             fillParameters(query, (PositionalParameters) params);
381         } else if (params instanceof NamedParameters) {
382             fillParameters(query, (NamedParameters) params);
383         } else {
384             throw new UnsupportedOperationException("不支持的参数形式");
385         }
386     }
387 
388     private void fillParameters(Query query, PositionalParameters params) {
389         Object[] paramArray = params.getParams();
390         for (int i = 0; i < paramArray.length; i++) {
391             query = query.setParameter(i, paramArray[i]);
392         }
393     }
394 
395     private void fillParameters(Query query, NamedParameters params) {
396         for (Map.Entry<String, Object> entry : params.getParams().entrySet()) {
397             Object value = entry.getValue();
398             if (value instanceof Collection) {
399                 query.setParameterList(entry.getKey(), (Collection) value);
400             } else if (value.getClass().isArray()) {
401                 query.setParameterList(entry.getKey(), (Object[]) value);
402             } else {
403                 query.setParameter(entry.getKey(), value);
404             }
405         }
