3   import org.dbunit.JdbcDatabaseTester;
4   import org.dbunit.database.DatabaseConfig;

6   import org.dbunit.dataset.datatype.DefaultDataTypeFactory;
7   import org.dbunit.ext.mysql.MySqlDataTypeFactory;
8   
9   public class MySQLJdbcDatabaseTester extends JdbcDatabaseTester {
10  
11  	public MySQLJdbcDatabaseTester(String driverClass, String connectionUrl)
12  			throws ClassNotFoundException {
13  		super(driverClass, connectionUrl);
14  	}
15  
16  	public MySQLJdbcDatabaseTester(String driverClass, String connectionUrl,
17  			String username, String password) throws ClassNotFoundException {
18  		super(driverClass, connectionUrl, username, password, null);
19  	}
20  
21  	public MySQLJdbcDatabaseTester(String driverClass, String connectionUrl,
22  			String username, String password, String schema)
23  			throws ClassNotFoundException {
24  		super(driverClass, connectionUrl, username, password, schema);
25  	}
26  
27  	@Override
28  	public IDatabaseConnection getConnection() throws Exception {
29  
30  		IDatabaseConnection conn = super.getConnection();
31  
32  		DefaultDataTypeFactory datatypeFactory = new MySqlDataTypeFactory();
33  		conn.getConfig().setProperty(DatabaseConfig.PROPERTY_DATATYPE_FACTORY,
34  				datatypeFactory);
35  
36  		return conn;
37  	}

39  }
