3   import com.google.gson.*;
4   import org.dayatang.utils.ObjectSerializer;
5   
6   import java.lang.reflect.Type;
7   import java.sql.Timestamp;
8   import java.util.Arrays;
9   import java.util.Date;
10  import java.util.List;

13   * Created by yyang on 14-9-16.

15  public class GsonObjectSerializer implements ObjectSerializer {
16  
17      private Gson gson;
18  
19      public GsonObjectSerializer() {
20          gson = createBuilder().create();
21      }
22  
23      public GsonObjectSerializer(String... excludedProperties) {
24          gson = createBuilder()
25                  .addSerializationExclusionStrategy(new FieldNameExclusionStrategy(excludedProperties))
26                  .create();
27  
28      }
29  
30      private GsonBuilder createBuilder() {
31          return new GsonBuilder()
32                  .registerTypeAdapter(Date.class, new DateSerializer())
33                  .registerTypeAdapter(Date.class, new DateDeserializer())
34                  .serializeNulls();
35      }
36  
37      @Override
38      public String serialize(Object anObject) {
39          return gson.toJson(anObject);
40      }
41  
42      @Override
43      public <T> T deserialize(String serializedString, Class<T> objectClass) {
44          return gson.fromJson(serializedString, objectClass);
45      }
46  
47      private class DateSerializer implements JsonSerializer<Date> {
48          public JsonElement serialize(Date source, Type typeOfSource, JsonSerializationContext context) {
49              return new JsonPrimitive(Long.toString(source.getTime()));
50          }
51      }
52  
53      private class DateDeserializer implements JsonDeserializer<Date> {
54          public Date deserialize(JsonElement json, Type typeOfTarget, JsonDeserializationContext context) throws JsonParseException {
55              long time = Long.parseLong(json.getAsJsonPrimitive().getAsString());
56              if (typeOfTarget == Date.class) {
57                  return new Date(time);
58              } else if (typeOfTarget == Timestamp.class) {
59                  return new Timestamp(time);
60              } else if (typeOfTarget == java.sql.Date.class) {
61                  return new java.sql.Date(time);
62              } else {
63                  throw new IllegalArgumentException(getClass() + " cannot deserialize to " + typeOfTarget);
64              }
65  
66  
67          }
68      }
69  
70      private static class FieldNameExclusionStrategy implements ExclusionStrategy {
71  
72          private List<String> fieldNames;
73  
74          private FieldNameExclusionStrategy(String... fieldNames) {
75              this.fieldNames = Arrays.asList(fieldNames);
76          }
77  
78          @Override
79          public boolean shouldSkipField(FieldAttributes fieldAttributes) {
80              return fieldNames.contains(fieldAttributes.getName());
81          }
82  
83          @Override
84          public boolean shouldSkipClass(Class<?> aClass) {
85              return false;
86          }
87      }
88  
89  }
