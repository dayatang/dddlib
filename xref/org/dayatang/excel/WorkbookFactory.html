3   import org.apache.poi.hssf.usermodel.HSSFWorkbook;
4   import org.apache.poi.ss.usermodel.Workbook;
5   import org.apache.poi.xssf.usermodel.XSSFWorkbook;
6   
7   import java.io.*;
8   
9   public class WorkbookFactory {
10  	
11  	private WorkbookFactory() {
12  		super();
13  	}
14  
15  	public static Workbook createWorkbook(File excelFile) {
16  		return createWorkbook(excelFile, Version.of(excelFile.getName()));
17  	}
18  
19  	public static Workbook createWorkbook(File excelFile, Version version) {
20  		try {
21  			return createWorkbook(new FileInputStream(excelFile), version);
22  		} catch (FileNotFoundException e) {
23  			throw new ExcelException("File " + excelFile.getPath() + " not exists.", e);
24  		}
25  	}
26  	
27  	public static Workbook createWorkbook(InputStream in, Version version) {
28  			try {
29  				if (version == Version.XLSX) {
30  					return new XSSFWorkbook(in);
31  				} else {
32  					return new HSSFWorkbook(in);
33  				}
34  			} catch (IOException e) {
35  				throw new ExcelException(e);
36  			}
37  	}
38  }
