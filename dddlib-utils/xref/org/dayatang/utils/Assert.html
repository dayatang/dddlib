24      /**
25       * 断言表达式{@code expression}的值为真。如果表达式的值为假则抛出
26       * <code>IllegalArgumentException</code>异常。
27       *
28       * <pre class="code">
29       * Assert.isTrue(i &gt; 0, &quot;The value must be greater than zero&quot;);
30       * </pre>
31       *
32       * @param expression 一个布尔表达式
33       * @param message 断言失败时呈现的异常消息
34       * @throws IllegalArgumentException 如果表达式的值是<code>false</code>
35       */
36      public static void isTrue(boolean expression, String message) {
37          if (!expression) {
38              throw new IllegalArgumentException(message);
39          }
40      }

657      * array is empty!
