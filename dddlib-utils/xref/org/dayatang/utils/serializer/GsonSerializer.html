15  public class GsonSerializer implements ObjectSerializer {
16  
17      private Gson gson;
18  
19      public GsonSerializer() {
20          this.gson = new GsonBuilder().create();
21      }
22  
23      public GsonSerializer(Gson gson) {
24          this.gson = gson;
25      }
26  
27      @Override
28      public String serialize(Object anObject) {
29          return gson.toJson(anObject);
30      }
31  
32      @Override
33      public <T> T deserialize(String serializedString, Class<T> objectClass) {
34          return gson.fromJson(serializedString, objectClass);
35      }
