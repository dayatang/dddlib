60          undergraduate = createDictionary("01", "本科", education, 100, "05");

65      private DictionaryCategory createCategory(String name, int sortOrder) {
66          DictionaryCategory category = new DictionaryCategory();
67          category.setName(name);
68          category.setSortOrder(sortOrder);
69          repository.save(category);
70          repository.flush();
71          return category;
72      }
73  
74      private Dictionary createDictionary(String code, String text, DictionaryCategory category, int sortOrder,
75              String parentCode) {
76          Dictionary dictionary = new Dictionary(code, text, category);
77          dictionary.setSortOrder(sortOrder);
78          dictionary.setParentCode(parentCode);
79          repository.save(dictionary);
80          repository.flush();
81          return dictionary;
82      }

84      @Test
85      public void testAddAndRemove() {
86          Dictionary dictionary = new Dictionary("2001", "双硕士", gender);
87          assertFalse(dictionary.existed());
88          dictionary = repository.save(dictionary);
89          assertTrue(dictionary.existed());
90          assertNotNull(dictionary.getId());
91          repository.remove(dictionary);
92          assertNull(repository.get(Dictionary.class, dictionary.getId()));
93      }

485         dictionaries = repository.findByExample(example, ExampleSettings.create(Dictionary.class).excludeZeroes().enableLike());
