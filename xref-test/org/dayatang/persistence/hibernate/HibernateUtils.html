3   import org.dayatang.persistence.test.domain.Dictionary;
4   import org.dayatang.persistence.test.domain.DictionaryCategory;
5   import org.hibernate.SessionFactory;
6   import org.hibernate.cfg.Configuration;
7   import org.hibernate.tool.hbm2ddl.SchemaExport;
8   
9   public class HibernateUtils {
10      private static SessionFactory sessionFactory = buildSessionFactory();
11  
12      @SuppressWarnings("deprecation")
13  	private static SessionFactory buildSessionFactory() {
14          try {
15              Configuration cfg = new Configuration()
16          		.addAnnotatedClass(DictionaryCategory.class)
17          		.addAnnotatedClass(Dictionary.class)
18          		.configure();
19          	new SchemaExport(cfg).create(false, true);
20              return cfg.buildSessionFactory();
21          }
22          catch (Exception ex) {
23              // Make sure you log the exception, as it might be swallowed
24              System.err.println("Initial SessionFactory creation failed." + ex);
25              throw new ExceptionInInitializerError(ex);
26          }
27         
28      }
29  
30      public static synchronized SessionFactory getSessionFactory() {
31      	if (sessionFactory == null || sessionFactory.isClosed()) {
32      		sessionFactory = buildSessionFactory();
33      	}
34          return sessionFactory;
35      }
36  
37      public static void close() {
38      	sessionFactory.close();
39      }
