33      @Before
34      public void setUp() throws Exception {
35          instance = new GsonObjectSerializer();
36          Date occurredOn = DateUtils.date(2002, 4, 11);
37          event = new DomainEventSub(occurredOn, 1);
38          event.setId("anId");
39          event.setProp1("abc");
40      }
41  
42      @Test
43      public void testSerialize() throws Exception {
44          String result = "{\"prop1\":\"abc\",\"prop2\":null,\"id\":\"anId\",\"occurredOn\":\"1018454400000\",\"version\":1}";
45          assertThat(instance.serialize(event), is(result));
46      }
47  
48      @Test
49      public void testSerializeExcludeField() throws Exception {
50          String result = "{\"prop1\":\"abc\",\"prop2\":null,\"id\":\"anId\",\"version\":1}";
51          instance = new GsonObjectSerializer("occurredOn");
52          assertThat(instance.serialize(event), is(result));
53      }
54  
55      @Test
56      public void testDeserialize() throws Exception {
57          String eventBody = "{\"prop1\":\"abc\",\"prop2\":null,\"id\":\"anId\",\"occurredOn\":\"1018454400000\",\"version\":1}";
58          DomainEventSub result = instance.deserialize(eventBody, DomainEventSub.class);
59          assertThat(result.getOccurredOn(), is(event.getOccurredOn()));
60          assertThat(result.getProp1(), is(event.getProp1()));
61          assertThat(result.getVersion(), is(event.getVersion()));
62          assertNull(result.getProp2());
63      }
