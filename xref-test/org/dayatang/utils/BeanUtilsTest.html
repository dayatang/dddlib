27      @Test
28      public void testGetPropTypes() {
29          Map<String, Class<?>> types = instance.getPropTypes();
30          assertEquals(int.class, types.get("id"));
31          assertEquals(String.class, types.get("name"));
32          assertEquals(boolean.class, types.get("disabled"));
33          assertEquals(double.class, types.get("price"));
34      }

36      @Test
37      public void testGetPropValues() {
38          Map<String, Object> values = instance.getPropValues();
39          assertEquals(1, values.get("id"));
40          assertEquals("abc", values.get("name"));
41          assertNull(values.get("disabled"));
42          assertEquals(12.5, (Double) values.get("price"), 0.001);
43      }

45      @Test
46      public void testGetPropNames() {

51      @Test
52      public void testGetReadablePropNames() {
53          Set<String> results = instance.getReadablePropNames();

56          assertFalse(results.contains("disabled"));
