1   package org.dayatang.dsmonitor.dao.impl;
2   
3   import org.dayatang.dsmonitor.dao.Dao;
4   import org.hibernate.Query;
5   import org.hibernate.Session;
6   import org.hibernate.SessionFactory;
7   
8   import javax.inject.Inject;
9   import java.util.List;
10  
11  @SuppressWarnings("rawtypes")
12  public class HibernateDaoImpl implements Dao {
13  	
14  	@Inject
15  	private SessionFactory sessionFactory;
16  	
17  	public void setSessionFactory(SessionFactory sessionFactory) {
18  		this.sessionFactory = sessionFactory;
19  	}
20  
21  	private Session getSession() {
22  		return sessionFactory.openSession();
23  	}
24  
25  	public List listResult(final String queryStr, final Object... values) {
26  		return createQuery(queryStr, values).list();
27  	}
28  
29  	public List listResultWithoutCloseConnection(String queryStr,
30  			Object... values) {
31  		return createQuery(queryStr, values).list();
32  	}
33  
34  	private Query createQuery(String queryStr, Object... values) {
35  		Query query = getSession().createQuery(queryStr);
36  		for (int i = 0; i < values.length; i++) {
37  			query.setParameter(i, values[i]);
38  		}
39  		return query;
40  	}
41  
42  }
