159     private class Item {
160 
161         private int id;
162         private String name;
163         private boolean disabled;
164 
165         private Item() {
166         }
167 
168         private Item(int id, String name, boolean disabled) {
169             this.id = id;
170             this.name = name;
171             this.disabled = disabled;
172         }
173 
174         public int getId() {
175             return id;
176         }
177 
178         public void setId(int id) {
179             this.id = id;
180         }
181 
182         public String getName() {
183             return name;
184         }
185 
186         public void setName(String name) {
187             this.name = name;
188         }
189 
190         public boolean isDisabled() {
191             return disabled;
192         }
193 
194         public void setDisabled(boolean disabled) {
195             this.disabled = disabled;
196         }
197 
198         @Override
199         public boolean equals(Object o) {
200             if (this == o) return true;
201             if (o == null || getClass() != o.getClass()) return false;
202 
203             Item item = (Item) o;
204 
205             if (disabled != item.disabled) return false;
206             if (id != item.id) return false;
207             if (!name.equals(item.name)) return false;
208 
209             return true;
210         }
211 
212         @Override
213         public int hashCode() {
214             int result = id;
215             result = 31 * result + name.hashCode();
216             result = 31 * result + (disabled ? 1 : 0);
217             return result;
218         }
219 
220         @Override
221         public String toString() {
222             return "Item{" +
223                     "id=" + id +
224                     ", name='" + name + '\'' +
225                     ", disabled=" + disabled +
226                     '}';
227         }
228     }
