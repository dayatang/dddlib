31      /**
32       * 设置了InstanceProvider，并且可通过InstanceProvider找到Bean
33       */
34      @Test
35      public void testGetInstanceByInstanceProvider() {
36          InstanceFactory.setInstanceProvider(instanceProvider);
37          when(instanceProvider.getInstance(Service.class)).thenReturn(service1);
38          when(instanceProvider.getInstance(Service.class, "service2")).thenReturn(service2);
39          when(instanceProvider.getInstance(Service.class, TheAnnotation.class)).thenReturn(service3);
40          assertEquals(service1, InstanceFactory.getInstance(Service.class));
41          assertEquals(service2, InstanceFactory.getInstance(Service.class, "service2"));
42          assertEquals(service3, InstanceFactory.getInstance(Service.class, TheAnnotation.class));
43      }

71       * 通过Bind方法注册的Bean
72       */
73      @Test
74      public void testGetInstanceByBinder() {
75          MyService21 service21 = new MyService21();
76          MyService22 service22 = new MyService22();
77          MyService23 service23 = new MyService23();
78          InstanceFactory.bind(Service2.class, service21);
79          assertNotNull(InstanceFactory.getInstance(Service2.class));
80          InstanceFactory.bind(Service2.class, service22, "service2");
81          InstanceFactory.bind(Service2.class, service23, "service3");
82          InstanceFactory.bind(Service2.class, service23, TheAnnotation.class);
83          assertEquals(service22, InstanceFactory.getInstance(Service2.class, "service2"));
84          assertEquals(service23, InstanceFactory.getInstance(Service2.class, "service3"));
85          assertEquals(service23, InstanceFactory.getInstance(Service2.class, TheAnnotation.class));
86      }
