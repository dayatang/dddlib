1   package org.dddlib.organisation.domain;
2   
3   import java.util.Date;
4   import java.util.List;
5   
6   import javax.persistence.DiscriminatorValue;
7   import javax.persistence.Entity;
8   import javax.persistence.NamedQueries;
9   import javax.persistence.NamedQuery;
10  
11  import org.apache.commons.lang3.builder.ToStringBuilder;
12  
13  @Entity
14  @DiscriminatorValue("Employment")
15  @NamedQueries({
16      @NamedQuery(name = "Employment.getEmployer", query = "select o.commissioner from Employment o where o.responsible = :employee and o.fromDate <= :date and o.toDate > :date"),
17      @NamedQuery(name = "Employment.getEmployees", query = "select o.responsible from Employment o where o.commissioner = :employer and o.fromDate <= :date and o.toDate > :date")})
18  public class Employment extends Accountability<Company, Employee> {
19  
20      private static final long serialVersionUID = 7390804525640459582L;
21  
22      protected Employment() {
23      }
24  
25      public Employment(Company company, Employee employee, Date date) {
26          super(company, employee, date);
27      }
28  
29      public static Company getEmployer(Employee employee, Date date) {
30          return getRepository().createNamedQuery("Employment.getEmployer")
31                  .addParameter("employee", employee).addParameter("date", date).singleResult();
32      }
33  
34      public static List<Employee> getEmployees(Company employer, Date date) {
35          return getRepository().createNamedQuery("Employment.getEmployees")
36                  .addParameter("employer", employer).addParameter("date", date).list();
37      }
38  
39      @Override
40      public String toString() {
41          return new ToStringBuilder(this).append(getCommissioner())
42                  .append(getResponsible()).build();
43      }
44  
45  }
