1   package org.dayatang.domain;
2   
3   import org.apache.commons.lang3.builder.EqualsBuilder;
4   import org.apache.commons.lang3.builder.HashCodeBuilder;
5   import org.dayatang.utils.Assert;
6   
7   /**
8    * 键值对数据类型
9    * @author yyang (<a href="mailto:gdyangyu@gmail.com">gdyangyu@gmail.com</a>)
10   */
11  public class KeyValue<K, V> {
12  	
13  	private K key;
14  	
15  	private V value;
16  
17  	/**
18  	 * 生成键值对
19  	 * @param key 键
20  	 * @param value 值
21  	 */
22  	public KeyValue(K key, V value) {
23  		Assert.notNull(key, "Key must not be null!");
24  		this.key = key;
25  		this.value = value;
26  	}
27  
28  	/**
29  	 * 获得键
30  	 * @return 键
31  	 */
32  	public K getKey() {
33  		return key;
34  	}
35  
36  	/**
37  	 * 获得值
38  	 * @return 值
39  	 */
40  	public V getValue() {
41  		return value;
42  	}
43  
44  	@Override
45  	public int hashCode() {
46  		return new HashCodeBuilder(17, 31).append(key).append(value).toHashCode();
47  	}
48  
49  	@SuppressWarnings("unchecked")
50  	@Override
51  	public boolean equals(Object other) {
52  		if (this == other) {
53  			return true;
54  		}
55  		if (!(other instanceof KeyValue)) {
56  			return false;
57  		}
58  		KeyValue<K, V> that = (KeyValue<K, V>) other;
59  		return new EqualsBuilder().append(this.key, that.key)
60  				.append(this.value, that.value).isEquals();
61  	}
62  
63  	@Override
64  	public String toString() {
65  		return "KeyValue [key=" + key + ", value=" + value + "]";
66  	}
67  
68  }
