1   package org.dayatang.domain.event;
2   
3   import java.util.Date;
4   import java.util.List;
5   
6   /**
7    * 事件存储接口
8    * Created by yyang on 14-9-15.
9    */
10  public interface EventStore {
11  
12      /**
13       * 获取指定时间范围发生的历史事件的集合，时间范围包含occurredFrom，不包含occurredTo。
14       *
15       * @param occurredFrom 事件发生时间的下限
16       * @param occurredTo   事件发生时间的上限
17       * @return 指定时间范围发生的事件的集合，按发生时间升序排序。
18       */
19      public List<StoredEvent> findStoredEventsBetween(Date occurredFrom, Date occurredTo);
20  
21      /**
22       * 获取指定时间及其之后发生的历史事件的集合。
23       *
24       * @param occurredFrom 事件发生时间的下限
25       * @return 指定时间及其之后发生的事件的集合，按发生时间升序排序。
26       */
27      public List<StoredEvent> findStoredEventsSince(Date occurredFrom);
28  
29      /**
30       * 向事件存储中插入一个新的领域事件
31       *
32       * @param domainEvent 一个领域事件
33       * @return 代表领域事件的存储的事件
34       */
35      public StoredEvent append(DomainEvent domainEvent);
36  
37      /**
38       * 关闭事件存储
39       */
40      public void close();
41  
42      /**
43       * 统计存储的事件的数量
44       *
45       * @return 已存储的事件的数量
46       */
47      public long countStoredEvents();
48  
49  }
