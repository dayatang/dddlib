3   import org.dayatang.domain.InstanceLocator;
4   
5   import java.lang.annotation.Annotation;
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   /**
10   * Created by yyang on 15/1/14.
11   */
12  public class MapInstanceLocator implements InstanceLocator {
13  
14      private Map<Object, Object> instances = new HashMap<Object, Object>();
15  
16      public MapInstanceLocator(Map<Object, Object> instances) {
17          this.instances = instances;
18      }
19  
20      @Override
21      public <T> T getInstance(Class<T> beanType) {
22          return (T) instances.get(beanType);
23      }
24  
25      @Override
26      public <T> T getInstance(Class<T> beanType, String beanName) {
27          return (T) instances.get(toName(beanType, beanName));
28      }
29  
30      @Override
31      public <T> T getInstance(Class<T> beanType, Class<? extends Annotation> annotationType) {
32          return (T) instances.get(toName(beanType, annotationType));
33      }
34  
35      private static String toName(Class<?> beanType, String beanName) {
36          return beanType.getName() + ":" + beanName;
37      }
38  
39      private static String toName(Class<?> beanType, Class<? extends Annotation> annotationType) {
40          return beanType.getName() + ":" + annotationType.getName();
41      }
42  }
