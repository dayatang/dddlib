446     @Test
447     public void testGetQueryString() {
448     	EqCriterion criterion1 = new EqCriterion("name", "abc");
449     	InCriterion criterion2 = new InCriterion("age", Arrays.asList(1, 2));
450         instance.eq("name", "abc")
451                         .isEmpty("post")
452                         .notNull("birthday")
453                         .in("age", Arrays.asList(1, 2))
454                         .getQueryString();        
455         assertEquals("select distinct(rootEntity) from org.dayatang.domain.entity.MyEntity as rootEntity  "
456                 + "where rootEntity.name = :rootEntity_name" + criterion1.hashCode() + " "
457                 + "and rootEntity.post is empty "
458                 + "and rootEntity.birthday is not null "
459                 + "and rootEntity.age in :rootEntity_age" + criterion2.hashCode(),
460                 instance.getQueryString());
461         assertEquals(NamedParameters.create()
462                 .add("rootEntity_name" + criterion1.hashCode(), "abc")
463                 .add("rootEntity_age" + criterion2.hashCode(), Arrays.asList(1, 2)),
464                 instance.getParameters());
465     }

467     @Test
468     public void testGetQueryString2() {
469     	EqCriterion criterion1 = new EqCriterion("name", "abc");
470     	InCriterion criterion2 = new InCriterion("age", Arrays.asList(1, 2));
471     	
472     	
473         assertEquals("select distinct(rootEntity) from org.dayatang.domain.entity.MyEntity as rootEntity  "
474                 + "where rootEntity.name = :rootEntity_name" + criterion1.hashCode() + " "
475                 + "and rootEntity.post is empty "
476                 + "and rootEntity.birthday is not null "
477                 + "and rootEntity.age in :rootEntity_age" + criterion2.hashCode() + " "
478                 + "order by rootEntity.name asc", 
479                 instance.eq("name", "abc")
480                         .isEmpty("post")
481                         .notNull("birthday")
482                         .in("age", Arrays.asList(1, 2))
483                         .asc("name")
484                         .getQueryString());
