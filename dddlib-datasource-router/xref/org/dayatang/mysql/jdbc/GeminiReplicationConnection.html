18  	public GeminiReplicationConnection(Properties masterProperties, Properties slaveProperties) throws SQLException {
19  		NonRegisteringDriver driver = new NonRegisteringDriver();
20  
21  		StringBuilder masterUrl = new StringBuilder("jdbc:mysql://");
22  		StringBuilder slaveUrl = new StringBuilder("jdbc:mysql://");
23  
24  		String masterHost = masterProperties.getProperty(NonRegisteringDriver.HOST_PROPERTY_KEY);
25  		if (masterHost != null) {
26  			masterUrl.append(masterHost);
27  		}
28  
29  		String slaveHost = slaveProperties.getProperty(NonRegisteringDriver.HOST_PROPERTY_KEY);
30  		if (slaveHost != null) {
31  			slaveUrl.append(slaveHost);
32  		}
33  
34  		String masterDb = masterProperties.getProperty(NonRegisteringDriver.DBNAME_PROPERTY_KEY);
35  		masterUrl.append("/");
36  		if (masterDb != null) {
37  			masterUrl.append(masterDb);
38  		}
39  
40  		String slaveDb = slaveProperties.getProperty(NonRegisteringDriver.DBNAME_PROPERTY_KEY);
41  		slaveUrl.append("/");
42  		if (slaveDb != null) {
43  			slaveUrl.append(slaveDb);
44  		}
45  
46  		slaveProperties.setProperty("roundRobinLoadBalance", "true");
47  
48  		this.masterConnection = (LoadBalancedMySQLConnection) driver.connect(masterUrl.toString(), masterProperties);
49  
50  		if (StringUtils.isBlank(slaveHost) && slaveUrl.toString().contains("///")) {
51  			info(" ----- the salveUrl contains the '///', that means there is no slaver, make slavesConnection = masterConnection --");
52  			slavesConnection = masterConnection;
53  		} else {
54  			this.slavesConnection = (LoadBalancedMySQLConnection) driver.connect(slaveUrl.toString(), slaveProperties);
55  			this.slavesConnection.setReadOnly(true);
56  		}
57  
58  		this.currentConnection = this.masterConnection;
59  	}
