12  public class GeminiNonRegisteringReplicationDriver extends NonRegisteringDriver {
13  	
14  	public GeminiNonRegisteringReplicationDriver() throws SQLException {
15  		super();
16  	}
17  
18  	/*
19  	 * (non-Javadoc)
20  	 * 
21  	 * @see java.sql.Driver#connect(java.lang.String, java.util.Properties)
22  	 */
23  	public Connection connect(String url, Properties info) throws SQLException {
24  		Properties parsedProps = parseURL(url, info);
25  
26  		if (parsedProps == null) {
27  			return null;
28  		}
29  
30  		Properties masterProps = (Properties) parsedProps.clone();
31  		Properties slavesProps = (Properties) parsedProps.clone();
32  
33  		// Marker used for further testing later on, also when
34  		// debugging
35  		slavesProps.setProperty("com.mysql.jdbc.ReplicationConnection.isSlave",
36  				"true");
37  
38  		String hostValues = parsedProps.getProperty(HOST_PROPERTY_KEY);
39  
40  		if (hostValues != null) {
41  			StringTokenizer st = new StringTokenizer(hostValues, ",");
42  
43  			StringBuffer masterHost = new StringBuffer();
44  			StringBuffer slaveHosts = new StringBuffer();
45  
46  			if (st.hasMoreTokens()) {
47  				String[] hostPortPair = parseHostPortPair(st.nextToken());
48  
49  				if (hostPortPair[HOST_NAME_INDEX] != null) {
50  					masterHost.append(hostPortPair[HOST_NAME_INDEX]);
51  				}
52  
53  				if (hostPortPair[PORT_NUMBER_INDEX] != null) {
54  					masterHost.append(":");
55  					masterHost.append(hostPortPair[PORT_NUMBER_INDEX]);
56  				}
57  			}
58  
59  			boolean firstSlaveHost = true;
60  
61  			while (st.hasMoreTokens()) {
62  				String[] hostPortPair = parseHostPortPair(st.nextToken());
63  
64  				if (!firstSlaveHost) {
65  					slaveHosts.append(",");
66  				} else {
67  					firstSlaveHost = false;
68  				}
69  
70  				if (hostPortPair[HOST_NAME_INDEX] != null) {
71  					slaveHosts.append(hostPortPair[HOST_NAME_INDEX]);
72  				}
73  
74  				if (hostPortPair[PORT_NUMBER_INDEX] != null) {
75  					slaveHosts.append(":");
76  					slaveHosts.append(hostPortPair[PORT_NUMBER_INDEX]);
77  				}
78  			}
79  
80  //			if (slaveHosts.length() == 0) {
81  //				throw SQLError
82  //						.createSQLException(
83  //								"Must specify at least one slave host to connect to for master/slave replication load-balancing functionality",
84  //								SQLError.SQL_STATE_INVALID_CONNECTION_ATTRIBUTE);
85  //			}
86  
87  			masterProps.setProperty(HOST_PROPERTY_KEY, masterHost.toString());
88  			slavesProps.setProperty(HOST_PROPERTY_KEY, slaveHosts.toString());
89  		}
90  
91  		return new GeminiReplicationConnection(masterProps, slavesProps);
92  	}
93

96  		return null;
97  	}
