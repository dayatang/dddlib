30  	public Dbunit() {
31  		// String jdbc_url = PropertiesUtil.JDBC_URL
32  		// + "&sessionVariables=FOREIGN_KEY_CHECKS=0";
33  
34  		String jdbc_url = PropertiesUtil.JDBC_URL;
35  		// System.setProperty(PropertiesBasedJdbcDatabaseTester.DBUNIT_SCHEMA,
36  		// "LPS");
37  		try {
38  			// databaseTester = new MySQLJdbcDatabaseTester(
39  			// PropertiesUtil.JDBC_DRIVER, jdbc_url,
40  			// PropertiesUtil.JDBC_USERNAME, PropertiesUtil.JDBC_PASSWD);
41  			ComboPooledDataSource ds = new ComboPooledDataSource();
42  			ds.setDriverClass(PropertiesUtil.JDBC_DRIVER);
43  			ds.setUser(PropertiesUtil.JDBC_USERNAME);
44  			ds.setPassword(PropertiesUtil.JDBC_PASSWD);
45  			ds.setJdbcUrl(jdbc_url);
46  
47  			if (PropertiesUtil.JDBC_DRIVER.toLowerCase().contains("mysql")) {
48  				logger.info("use mysql datasource tester.");
49  				jdbc_url = PropertiesUtil.JDBC_URL
50  						+ "&sessionVariables=FOREIGN_KEY_CHECKS=0";
51  				ds.setJdbcUrl(jdbc_url);
52  				databaseTester = new MySQLDataSourceDatabaseTester(ds);
53  			} else if (PropertiesUtil.JDBC_DRIVER.toLowerCase().contains(
54  					"oracle")) {
55  				logger.info("use oracle datasource tester.");
56  				databaseTester = new OracleDataSourceDatabaseTester(ds,
57  						PropertiesUtil.JDBC_USERNAME);
58  			} else if (PropertiesUtil.JDBC_DRIVER.toLowerCase().contains("h2")) {
59  				logger.info("use h2 datasource tester.");
60  				databaseTester = new H2DataSourceDatabaseTester(ds);
61  			} else {
62  				throw new RuntimeException("不支持的数据库类型！");
63  			}
64  
65  		} catch (Exception ex) {
66  			logger.error("构造测试环境失败！");
67  			ex.printStackTrace();
68  		}
69  
70  	}

72  	@Before
73  	public void setUp() throws Exception {
74  		if (existsDataFiles()) {
75  			// initialize your dataset here
76  
77  			String[] dataSetPaths = getDataSetFilePaths();
78  
79  			IDataSet[] dataSet = new IDataSet[dataSetPaths.length];
80  
81  			for (int i = 0; i < dataSetPaths.length; i++) {
82  
83  				String path = dataSetPaths[i].startsWith("/") ? dataSetPaths[i]
84  						: "/" + dataSetPaths[i];
85  				// dataSet[i] = new FlatXmlDataSet(DbUnit.class
86  				// .getResourceAsStream(path));
87  				dataSet[i] = getDataSetObject(path);
88  				logger.debug("载入数据库资源文件：" + path);
89  
90  			}
91  
92  			CompositeDataSet compositeDateSet = new CompositeDataSet(
93  					(IDataSet[]) dataSet);
94  
95  			databaseTester.setDataSet(compositeDateSet);
96  			// will call default setUpOperation
97  			databaseTester.setSetUpOperation(setUpOp());
98  			databaseTester.onSetup();
99  		} else {
100 			logger.warn("没有指定数据集！");
101 		}
102 
103 	}
104 
105 	private IDataSet getDataSetObject(String path) throws Exception {
106 		if (getDataSetStrategy().equals(DataSetStrategy.Xml)) {
107 			return new XmlDataSet(Dbunit.class.getResourceAsStream(path));
108 		} else if (getDataSetStrategy().equals(DataSetStrategy.FlatXml)) {
109 			boolean enableColumnSensing = true;
110 			InputStreamReader inReader = new InputStreamReader(Dbunit.class
111 					.getResourceAsStream(path), "UTF-8");
112 			// FlatXmlDataSet fxset = new FlatXmlDataSet(inReader, true,
113 			// enableColumnSensing, false);
114 			// return fxset;
115 			return new CachedDataSet(new FlatXmlProducer(new InputSource(
116 					inReader), true, enableColumnSensing, false));
117 
118 		} else {
119 			return new XmlDataSet(Dbunit.class.getResourceAsStream(path));
120 		}
121 	}
