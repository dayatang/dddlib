30  	private Module createModule() {
31  		return new Module() {
32  			@Override
33  			public void configure(Binder binder) {
34  				binder.bind(Service2.class).to(MyService21.class).in(Scopes.SINGLETON);
35  				binder.bind(Service.class).annotatedWith(Names.named("service2")).to(MyService2.class).in(Scopes.SINGLETON);
36  				binder.bind(Service.class).annotatedWith(Names.named("service1")).to(MyService1.class).in(Scopes.SINGLETON);
37  				binder.bind(Service3.class).annotatedWith(MyBindingAnnotation.class).to(MyService31.class).in(Scopes.SINGLETON);
38  			}
39  		};
40  	}
41  
42  	@Override
43  	protected InstanceProvider createInstanceProvider() {
